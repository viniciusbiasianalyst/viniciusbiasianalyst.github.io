-- receita acumulada por canal com ranking mensal
WITH vendas_mensais AS (
  SELECT DATE_TRUNC('month', data_venda) AS mes,
    canal, SUM(valor) AS receita
  FROM pipeline_comercial
  WHERE status = 'convertido'
  GROUP BY 1, canal
),
ranking_canal AS (
  SELECT *,
    SUM(receita) OVER (PARTITION BY canal ORDER BY mes) AS acumulada,
    RANK() OVER (PARTITION BY mes ORDER BY receita DESC) AS pos
  FROM vendas_mensais
)
SELECT * FROM ranking_canal
WHERE pos <= 3 ORDER BY mes DESC, pos;
-- SLA médio por agente com joins em múltiplas tabelas
SELECT
  a.nome AS agente, t.nome_time,
  COUNT(c.id) AS total_tickets,
  AVG(c.tempo_resolucao) AS sla_medio,
  SUM(CASE WHEN c.csat >= 4 THEN 1 ELSE 0 END)
    / NULLIF(COUNT(c.id), 0) AS taxa_sat
FROM chamados c
INNER JOIN agentes a ON c.agente_id = a.id
LEFT JOIN times t ON a.time_id = t.id
WHERE c.criado_em >= NOW() - INTERVAL '30 days'
GROUP BY a.nome, t.nome_time
HAVING COUNT(c.id) > 5
ORDER BY sla_medio ASC;
-- segmentação de clientes: ativo, em risco e churn
WITH ultima_compra AS (
  SELECT cliente_id,
    MAX(data_pedido) AS ultimo_pedido,
    COUNT(*) AS total_pedidos
  FROM pedidos GROUP BY cliente_id
),
segmentos AS (
  SELECT u.cliente_id, c.nome, c.plano,
    DATE_PART('day', NOW() - u.ultimo_pedido) AS dias,
    CASE
      WHEN DATE_PART('day',NOW()-u.ultimo_pedido) < 30 THEN 'ativo'
      WHEN DATE_PART('day',NOW()-u.ultimo_pedido) < 90 THEN 'em risco'
      ELSE 'churn'
    END AS status_cliente
  FROM ultima_compra u
  JOIN clientes c ON u.cliente_id = c.id
)
SELECT status_cliente, COUNT(*) AS qtd,
  ROUND(AVG(dias),1) AS media_dias
FROM segmentos GROUP BY status_cliente;
CTE + Window Function: receita acumulada e ranking por canal com RANK() OVER PARTITION.

Exemplo de query — BigQuery
-- eventos GA4: sessões, usuários e taxa de engajamento
WITH base AS (
  SELECT
    DATE(TIMESTAMP_MICROS(event_timestamp)) AS dia,
    geo.country AS pais,
    device.category AS dispositivo,
    user_pseudo_id,
    MAX(CASE WHEN event_name = 'session_start' THEN 1 ELSE 0 END) AS teve_sessao,
    MAX(CASE WHEN event_name = 'user_engagement' THEN 1 ELSE 0 END) AS engajou
  FROM `projeto.analytics_123.events_*`
  WHERE _TABLE_SUFFIX BETWEEN '20250101' AND '20250331'
  GROUP BY dia, pais, dispositivo, user_pseudo_id
)
SELECT
  dia, pais, dispositivo,
  COUNT(DISTINCT user_pseudo_id) AS usuarios,
  SUM(teve_sessao) AS sessoes,
  ROUND(SUM(engajou) / NULLIF(SUM(teve_sessao),0) * 100, 1) AS taxa_engajamento
FROM base
GROUP BY dia, pais, dispositivo
ORDER BY dia DESC, usuarios DESC;
